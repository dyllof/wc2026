const entriesUrl = 'https://docs.google.com/spreadsheets/d/1XtiI0grU2Ebpz3PMOe1iFpVFu9Oh-w3S_yMcQoR6DXw/gviz/tq?tqx=out:csv&sheet=Entries'; const scoresUrl = 'https://docs.google.com/spreadsheets/d/1XtiI0grU2Ebpz3PMOe1iFpVFu9Oh-w3S_yMcQoR6DXw/gviz/tq?tqx=out:csv&sheet=TeamScores'; function parseCSV(text) { return text.trim().split('\n').map(r => r.split(',')); } async function loadData() { try { const [entriesText, scoresText] = await Promise.all([ fetch(entriesUrl).then(r => r.text()), fetch(scoresUrl).then(r => r.text()) ]); const entries = parseCSV(entriesText); const scores = parseCSV(scoresText); // Build score map const scoreMap = {}; scores.slice(1).forEach(row => { scoreMap[row[0]] = Number(row[1]) || 0; }); const multipliers = [6, 5, 4, 3, 2, 1]; const leaderboard = []; entries.slice(1).forEach(row => { const player = row[0]; let total = 0; for (let i = 0; i < 6; i++) { const team = row[i + 1]; total += (scoreMap[team] || 0) * multipliers[i]; } leaderboard.push({ player, total }); }); leaderboard.sort((a, b) => b.total - a.total); // Render leaderboard let html = ` `; leaderboard.forEach((p, i) => { let cls = ''; if (i === 0) cls = 'rank1'; if (i === 1) cls = 'rank2'; if (i === 2) cls = 'rank3'; html += ` `; }); html += '
RankPlayerScore
${i + 1} ${p.player} ${p.total}
'; document.getElementById('leaderboard').innerHTML = html; // Render team scores let teamHtml = ` `; scores.slice(1).forEach(row => { teamHtml += ` `; }); teamHtml += '
TeamPoints
${row[0]} ${row[1]}
'; document.getElementById('teams').innerHTML = teamHtml; } catch (err) { console.error(err); document.getElementById('leaderboard').innerHTML = "Error loading data"; document.getElementById('teams').innerHTML = "Error loading data"; } } loadData();
